#include<iostream>
using namespace std;

int Path(int n)
{
	if(n==1) return 1;
	if(n==2) return 1;
	if(n%2==1) return Path(n-1) + Path(n-2);
	return Path(n-1) + Path(n-2) + Path(n-3); 
}
