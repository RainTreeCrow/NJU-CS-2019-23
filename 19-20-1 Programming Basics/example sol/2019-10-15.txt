// 问题1
#include<iostream>
#include<cmath>
using namespace std;

const double ZERO = 0.000001;

int factorial(int n)
{
	if(n == 1) return 1;
	return n * factorial(n-1);
}

int main()
{
	double x;
	cin>>x;
	
	double series_sum = x;
	
	for(int i = 1;; ++ i){
		double series = pow(-1, i) * pow(x, 2*i+1) / ((2*i+1)*factorial(i));
		series_sum += series;
		//cout<<"i="<<i<<" "<<series<<" "<<series_sum<<endl;
		if(fabs(series) < ZERO) break;
	}
	
	cout<<series_sum<<endl;
	
	// Used for debug 
	//cout<<x - pow(x,3)/3.0 + pow(x,5)/10.0 - pow(x,7)/42.0 + pow(x,9)/9/24 -pow(x,11)/11/24/5 + pow(x,13)/13/24/5/6<<endl;  
	
	return 0;
}

	return 0;
} 

void CalArea(double a,double b,double c) 
{ 
	double s=(a+b+c)/2.0; 
	double area=sqrt(s*(s-a)*(s-b)*(s-c)); 
	cout<<area<<endl; 
}

// 问题5
#include<iostream>
using namespace std;

int inverse(int n)
{
	int s=0;
	while (n){
		s = 10*s+n%10;
		n/=10;
	}
	return s;
}

int main()
{
    int m,n;
	cin >> n;
	int i=0;
	while (1)
	{
		m = inverse(n);
		if (m==n) break;
		n+=m;
		i++;
	}
	cout <<i<<"*"<<n<<endl;
	return 0;
}

int main()
{
	bool a[100000] = { 0 };
	int n;
	cin >> n;
	for (int i = 1; i <= n; i++)
	{
		for (int j = i; j <= n; j += i)  //这里不能写j++,会超时
		{
			a[j] = 1 - a[j];
		}
	}
	int sum = 0;
	for (int k = 1; k <= n; k++)
	{
		sum = sum + a[k];
	}
	cout << sum;
	return 0;
}

//问题7
#include<iostream>
using namespace std;
int fn(long long int n){      //这里的声明也必须是longlongint
	int step=0;
	while (n != 1) {
		if (n % 2 == 0) n = n / 2;
		else n = 3 * n + 1;
		step += 1;
	}
	return step;
}
int main() {
	long long int i,j; //如果用int会越界
	cin >> i>>j;
	int max=0;
	for (int k = i; k <= j; k++) {
		if (fn(k) > max) max = fn(k);
	}
	cout << max;
	return 0;
}
