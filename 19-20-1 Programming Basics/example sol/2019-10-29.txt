第一题
#include<iostream>
#include<cmath> 
using namespace std;

void CalArea(double a,double b,double c); 

int main() 
{ 
	double a,b,c;
	cin>>a>>b>>c; 
	CalArea(a,b,c); 
	return 0;
} 

void CalArea(double a,double b,double c) 
{ 
	double s=(a+b+c)/2.0; 
	double area=sqrt(s*(s-a)*(s-b)*(s-c)); 
	cout<<area<<endl; 
}

{
	char str[1000];

	int count = 0;

	{
		if (str[i] != ' ')
		{
			count++;
		}
		if (str[i] >= 'a'&&str[i] <= 'z')

			str[i] -= 'a' - 'A';
		}
	}

#include<iostream>
#include<math.h> 
using namespace std;
# define EPS 1.0e-8
double newton_sqrt(double x);
int main()
{
	double x, y;
	cin >> x;
	y = newton_sqrt(x);
	cout << y << endl;
	return 0;
}
double newton_sqrt(double x)
{
	double x0, x1;
	x1 = 1.0;
	do
	{
		x0 = x1;
		x1 = (x0 + x / x0) / 2;
	} while (fabs((x0 - x1) / x1) >= EPS);
	return x1;
}

第四题
#include<iostream>
#include<string.h>
using namespace std;
int FindChar(char s[], char ch, int n = 1);

int main()
{
	int n, p;
	char s[200], ch;
	cin.getline(s, 199);
	cin >> ch;
	p = FindChar(s, ch);
	cout << p << endl;
	n = 1;
	while (p)
	{
		p = FindChar(s, ch, p + 1);
		if (p) n++;
	}
	cout << n << endl;
	return 0;
}

int FindChar(char s[], char ch, int n)
{
	int i, len = strlen(s);
	if (n>len) return 0;
	for (i = n - 1; i<len; i++)
		if (s[i] == ch) return i + 1;
	return 0;
}

void DelRepCharacter(char *s);

int main()
{
	char s[200];
	cin.getline(s, 199);
	DelRepCharacter(s);
	//cout << s <<endl;
	return 0;
}

void DelRepCharacter(char *s)
{
	int i, j, n, len;
	len = strlen(s);
	char snew[200];
	for (i = 0; i<len; i++)
	{
		if (!s[i]) continue;
		for (j = i + 1; j<len; j++)
		{
			if (s[j] == s[i])
				s[j] = 0;
		}
	}
	for (n = i = 0; i<len; i++)
	{
		if (s[i])
		{
			snew[n] = s[i];
			n++;
		}
	}
	snew[n] = 0;
	cout << snew << endl;
}
