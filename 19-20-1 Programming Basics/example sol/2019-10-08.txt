int main()
{
	int a, b, t;
	cin >> a >> b;

	if (a<b)
	{
		t = a;

	}

// 问题4
#include <iostream>
using namespace std;

int main()
{
	int eng = 0, num = 0, oth = 0, spa = 0;
	for (int i = 1; i <= 10; i++)
	{
		char c = getchar();
		if ((c >= '0') && (c <= '9'))
		{
			num++;
		}
		else if ((c >= 'a') && (c <= 'z'))
		{
			eng++;
		}
		else if ((c >= 'A') && (c <= 'Z'))
		{
			eng++;
		}
		else if (c == ' ')
		{
			spa++;
		}
	}
	oth = 10 - eng - num - spa;
	cout << eng << endl << num << endl << spa << endl << oth << endl;
	return 0;
}
